49 _ODP_INLINE int odp_event_type_multi(const odp_event_t event[], int num,
50  odp_event_type_t *type_out)
51 {
52  int i;
53  odp_event_type_t type = __odp_event_type_get(event[0]);
54 
55  for (i = 1; i < num; i++) {
56  if (__odp_event_type_get(event[i]) != type)
57  break;
58  }
59 
60  *type_out = type;
61 
62  return i;
63 }
