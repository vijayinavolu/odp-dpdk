85 static inline uint64_t _odp_time_hw_to_ns(odp_time_t time)
86 {
87  uint64_t nsec;
88  uint64_t freq_hz = _odp_time_glob.hw_freq_hz;
89  uint64_t count = time.count;
90  uint64_t sec = 0;
91 
92  if (count >= freq_hz) {
93  sec = count / freq_hz;
94  count = count - sec * freq_hz;
95  }
96 
97  nsec = (_ODP_TIME_GIGA_HZ * count) / freq_hz;
98 
99  return (sec * _ODP_TIME_GIGA_HZ) + nsec;
100 }

102 static inline uint64_t _odp_time_convert_to_ns(odp_time_t time)
103 {
104  if (_odp_time_glob.use_hw)
105  return _odp_time_hw_to_ns(time);
106 
107  return time.nsec;
108 }
